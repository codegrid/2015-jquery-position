offset()してからshow()する

$('.run01').on('click', function(){
  $('.block').hide();
  setTimeout(function(){
    $('.block').offset({top:100,left:100}).show();
  },100);
});

クリックのつど、位置がずれていく

show()してからoffset()する

$('.run02').on('click', function(){
  $('.block').hide();
  setTimeout(function(){
    $('.block').show().offset({top:100,left:100});
  },100);
});
